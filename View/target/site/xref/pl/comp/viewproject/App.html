12  public class App extends Application {
13      @Override
14      /* modify the method declaration to throw generic Exception (in case any of the steps fail) */
15      public void start(Stage primaryStage) throws Exception {
16          FXMLLoader loader = new FXMLLoader();
17          loader.setResources(ResourceBundle.getBundle("lang.App", new Locale("pl")));
18          /* load layout.fxml from file and assign it to a scene root object */
19  
20          Parent root = loader.load(getClass().getResourceAsStream("MenuWindow.fxml"));
21  
22  
23          /* assign the root to a new scene and define its dimensions */
24          Scene scene = new Scene(root, 760, 560);
25  
26          /* set the title of the stage (window) */
27          primaryStage.setTitle("Sudoku");
28          /* set the scene of the stage to our newly created from the layout scene */
29          primaryStage.setScene(scene);
30          /* show the stage */
31          primaryStage.show();
32      }
33  
34      public static void main(String[] args) {
35          launch();
36      }
